62  The ReferenceGraph illustrates the references between a collection of objects
63  by rendering a directed graph. That requires that 'graphviz' is installed.
64 
65  >>> from pympler.refgraph import ReferenceGraph
66  >>> a = 42
67  >>> b = 'spam'
68  >>> c = {a: b}

70  >>> gb.render('spam.eps')

92  def _eliminate_leafs(self, graph):
93  """
94  Eliminate leaf objects - that are objects not referencing any other
95  objects in the list `graph`. Returns the list of objects without the
96  objects identified as leafs.
97  """
98  result = []
99  idset = set([id(x) for x in graph])
100  for n in graph:
101  refset = set([id(x) for x in get_referents(n)])
102  if refset.intersection(idset):
103  result.append(n)
104  return result

107  def _reduce_to_cycles(self):
108  """
109  Iteratively eliminate leafs to reduce the set of objects to only those
110  that build cycles. Return the number of objects involved in reference
111  cycles. If there are no cycles, `self.objects` will be an empty list and
112  this method returns 0.
113  """
114  cycles = self.objects[:]
115  cnt = 0
116  while cnt != len(cycles):
117  cnt = len(cycles)
118  cycles = self._eliminate_leafs(cycles)
119  self.objects = cycles
120  return len(self.objects)

123  def reduce_to_cycles(self):
124  """
125  Iteratively eliminate leafs to reduce the set of objects to only those
126  that build cycles. Return the reduced graph. If there are no cycles,
127  None is returned.
128  """
129  if not self._reduced:
130  reduced = copy(self)
131  reduced.objects = self.objects[:]
132  reduced.metadata = []
133  reduced.edges = []
134  self.num_in_cycles = reduced._reduce_to_cycles()
135  reduced.num_in_cycles = self.num_in_cycles
136  if self.num_in_cycles:
137  reduced._get_edges()
138  reduced._annotate_objects()
139  for meta in reduced.metadata:
140  meta.cycle = True
141  else:
142  reduced = None
143  self._reduced = reduced
144  return self._reduced

147  def _get_edges(self):
148  """
149  Compute the edges for the reference graph.
150  The function returns a set of tuples (id(a), id(b), ref) if a
151  references b with the referent 'ref'.
152  """
153  idset = set([id(x) for x in self.objects])
154  self.edges = set([])
155  for n in self.objects:
156  refset = set([id(x) for x in get_referents(n)])
157  for ref in refset.intersection(idset):
158  label = ''
159  members = None
160  if isinstance(n, dict):
161  members = n.items()
162  if not members:
163  members = named_refs(n)
164  for (k, v) in members:
165  if id(v) == ref:
166  label = k
167  break
168  self.edges.add(_Edge(id(n), ref, label))

171  def _annotate_groups(self):
172  """
173  Annotate the objects belonging to separate (non-connected) graphs with
174  individual indices.
175  """
176  g = {}
177  for x in self.metadata:
178  g[x.id] = x
179 
180  idx = 0
181  for x in self.metadata:
182  if not hasattr(x, 'group'):
183  x.group = idx
184  idx += 1
185  neighbors = set()
186  for e in self.edges:
187  if e.src == x.id:
188  neighbors.add(e.dst)
189  if e.dst == x.id:
190  neighbors.add(e.src)
191  for nb in neighbors:
192  g[nb].group = min(x.group, getattr(g[nb], 'group', idx))
193 
194  # Assign the edges to the respective groups. Both "ends" of the edge
195  # should share the same group so just use the first object's group.
196  for e in self.edges:
197  e.group = g[e.src].group
198 
199  self._max_group = idx

202  def _filter_group(self, group):
203  """
204  Eliminate all objects but those which belong to `group`.
205  ``self.objects``, ``self.metadata`` and ``self.edges`` are modified.
206  Returns `True` if the group is non-empty. Otherwise returns `False`.
207  """
208  self.metadata = [x for x in self.metadata if x.group == group]
209  group_set = set([x.id for x in self.metadata])
210  self.objects = [obj for obj in self.objects if id(obj) in group_set]
211  self.count = len(self.metadata)
212  if self.metadata == []:
213  return False
214 
215  self.edges = [e for e in self.edges if e.group == group]
216 
217  del self._max_group
218 
219  return True

222  def split(self):
223  """
224  Split the graph into sub-graphs. Only connected objects belong to the
225  same graph. `split` yields copies of the Graph object. Shallow copies
226  are used that only replicate the meta-information, but share the same
227  object list ``self.objects``.
228 
229  >>> from pympler.refgraph import ReferenceGraph
230  >>> a = 42
231  >>> b = 'spam'
232  >>> c = {a: b}
233  >>> t = (1,2,3)
234  >>> rg = ReferenceGraph([a,b,c,t])
235  >>> for subgraph in rg.split():
236  ... print subgraph.index
237  0
238  1
239  """
240  self._annotate_groups()
241  index = 0
242 
243  for group in range(self._max_group):
244  subgraph = copy(self)
245  subgraph.metadata = self.metadata[:]
246  subgraph.edges = self.edges.copy()
247 
248  if subgraph._filter_group(group):
249  subgraph.total_size = sum([x.size for x in subgraph.metadata])
250  subgraph.index = index
251  index += 1
252  yield subgraph

267  def _annotate_objects(self):
268  """
269  Extract meta-data describing the stored objects.
270  """
271  self.metadata = []
272  sizer = Asizer()
273  sizes = sizer.asizesof(*self.objects)
274  self.total_size = sizer.total
275  for obj, sz in zip(self.objects, sizes):
276  md = _MetaObject()
277  md.size = sz
278  md.id = id(obj)
279  try:
280  md.type = obj.__class__.__name__
281  except (AttributeError, ReferenceError): # pragma: no cover

287  def _get_graphviz_data(self):
288  """
289  Emit a graph representing the connections between the objects described
290  within the metadata list. The text representation can be transformed to
291  a graph with graphviz. Returns a string.
292  """
293  s = []
294  header = '// Process this file with graphviz\n'
295  s.append( header)
296  s.append('digraph G {\n')
297  s.append(' node [shape=box];\n')
298  for md in self.metadata:
299  label = trunc(md.str, 48).replace('"', "'")
300  extra = ''
301  if md.type == 'instancemethod':
302  extra = ', color=red'
303  elif md.type == 'frame':
304  extra = ', color=orange'
305  s.append(' "X%s" [ label = "%s\\n%s" %s ];\n' % \
306  (hex(md.id)[1:], label, md.type, extra))
307  for e in self.edges:
308  extra = ''
309  if e.label == '__dict__':
310  extra = ',weight=100'
311  s.append(' X%s -> X%s [label="%s"%s];\n' % \
312  (hex(e.src)[1:], hex(e.dst)[1:], e.label, extra))
313 
314  s.append('}\n')
315  return "".join(s)

318  def render(self, filename, cmd='dot', format='ps', unflatten=False):
319  """
320  Render the graph to `filename` using graphviz. The graphviz invocation
321  command may be overriden by specifying `cmd`. The `format` may be any
322  specifier recognized by the graph renderer ('-Txxx' command). The graph
323  can be preprocessed by the *unflatten* tool if the `unflatten` parameter
324  is True. If there are no objects to illustrate, the method does not
325  invoke graphviz and returns False. If the renderer returns successfully
326  (return code 0), True is returned.
327 
328  An `OSError` is raised if the graphviz tool cannot be found.
329  """
330  if self.objects == []:
331  return False
332 
333  data = self._get_graphviz_data()
334 
335  options = ('-Nfontsize=10',
336  '-Efontsize=10',
337  '-Nstyle=filled',
338  '-Nfillcolor=#E5EDB8',
339  '-Ncolor=#CCCCCC')
340  cmdline = (cmd, '-T%s' % format, '-o', filename) + options
341 
342  if unflatten:
343  p1 = Popen(('unflatten', '-l7'), stdin=PIPE, stdout=PIPE,
344  **popen_flags)
345  p2 = Popen(cmdline, stdin=p1.stdout, **popen_flags)
346  p1.communicate(encode4pipe(data))
347  p2.communicate()
348  return p2.returncode == 0
349  else:
350  p = Popen(cmdline, stdin=PIPE, **popen_flags)
351  p.communicate(encode4pipe(data))
352  return p.returncode == 0
