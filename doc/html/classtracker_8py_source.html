63  def __init__(self, instance, resolution_level=0, trace=False):
64  """
65  Create a weak reference for 'instance' to observe an object but which
66  won't prevent its deletion (which is monitored by the finalize
67  callback). The size of the object is recorded in 'snapshots' as
68  (timestamp, size) tuples.
69  """
70  self.ref = weakref_ref(instance, self.finalize)
71  self.id = id(instance)
72  self.repr = ''
73  self.name = str(instance.__class__)

358  def track_object(self, instance, name=None, resolution_level=0, keep=False, trace=False):
359  """
360  Track object 'instance' and sample size and lifetime information.
361  Not all objects can be tracked; trackable objects are class instances and
362  other objects that can be weakly referenced. When an object cannot be
363  tracked, a `TypeError` is raised.
364 
365  :param resolution_level: The recursion depth up to which referents are
366  sized individually. Resolution level 0 (default) treats the object
367  as an opaque entity, 1 sizes all direct referents individually, 2
368  also sizes the referents of the referents and so forth.
369  :param keep: Prevent the object's deletion by keeping a (strong)
370  reference to the object.
371  """
372 
373  # Check if object is already tracked. This happens if track_object is
374  # called multiple times for the same object or if an object inherits
375  # from multiple tracked classes. In the latter case, the most
376  # specialized class wins. To detect id recycling, the weak reference
377  # is checked. If it is 'None' a tracked object is dead and another one
378  # takes the same 'id'.
379  if id(instance) in self.objects and \
380  self.objects[id(instance)].ref() is not None:
381  return
382 
383  tobj = TrackedObject(instance, resolution_level=resolution_level, trace=trace)
384 
385  if name is None:
386  name = instance.__class__.__name__
387  if not name in self.index:
388  self.index[name] = []
389  self.index[name].append(tobj)
390  self.objects[id(instance)] = tobj
391 
392  if keep:
393  self._keepalive.append(instance)

396  def track_class(self, cls, name=None, resolution_level=0, keep=False, trace=False):
397  """
398  Track all objects of the class `cls`. Objects of that type that already
399  exist are *not* tracked. If `track_class` is called for a class already
400  tracked, the tracking parameters are modified. Instantiation traces can be
401  generated by setting `trace` to True.
402  A constructor is injected to begin instance tracking on creation
403  of the object. The constructor calls `track_object` internally.
404 
405  :param cls: class to be tracked, may be an old-style or a new-style class
406  :param name: reference the class by a name, default is the concatenation of
407  module and class name
408  :param resolution_level: The recursion depth up to which referents are
409  sized individually. Resolution level 0 (default) treats the object
410  as an opaque entity, 1 sizes all direct referents individually, 2
411  also sizes the referents of the referents and so forth.
412  :param keep: Prevent the object's deletion by keeping a (strong)
413  reference to the object.
414  :param trace: Save instantiation stack trace for each instance
415  """
416  if not isclass(cls):
417  raise TypeError("only class objects can be tracked")
418  if name is None:
419  name = cls.__module__ + '.' + cls.__name__
420  if self._is_tracked(cls):
421  self._track_modify(cls, name, resolution_level, keep, trace)
422  else:
423  self._inject_constructor(cls, self._tracker, name, resolution_level, keep, trace)

496  def create_snapshot(self, description='', compute_total=False):
497  """
498  Collect current per instance statistics and saves total amount of
499  memory associated with the Python process.
500 
501  If `compute_total` is `True`, the total consumption of all objects
502  known to *asizeof* is computed. The latter might be very slow if many
503  objects are mapped into memory at the time the snapshot is taken.
504  Therefore, `compute_total` is set to `False` by default.
505 
506  The overhead of the `ClassTracker` structure is also computed.
507 
508  Snapshots can be taken asynchronously. The function is protected with a
509  lock to prevent race conditions.
510  """
511 
512  try:
513  # TODO: It is not clear what happens when memory is allocated or
514  # released while this function is executed but it will likely lead
515  # to inconsistencies. Either pause all other threads or don't size
516  # individual objects in asynchronous mode.
517  self.snapshot_lock.acquire()
518 
519  timestamp = _get_time()
520 
521  sizer = asizeof.Asizer()
522  objs = [tobj.ref() for tobj in list(self.objects.values())]
523  sizer.exclude_refs(*objs)
524 
525  # The objects need to be sized in a deterministic order. Sort the
526  # objects by its creation date which should at least work for non-parallel
527  # execution. The "proper" fix would be to handle shared data separately.
528  tracked_objects = list(self.objects.values())
529  tracked_objects.sort(key=lambda x: x.birth)
530  for tobj in tracked_objects:
531  tobj.track_size(timestamp, sizer)
532 
533  snapshot = Snapshot()
534 
535  snapshot.timestamp = timestamp
536  snapshot.tracked_total = sizer.total
537  if compute_total:
538  snapshot.asizeof_total = asizeof.asizeof(all=True, code=True)
539  snapshot.system_total = pympler.process.ProcessMemoryInfo()
540  snapshot.desc = str(description)
541 
542  # Compute overhead of all structures, use sizer to exclude tracked objects(!)
543  snapshot.overhead = 0
544  if snapshot.tracked_total:
545  snapshot.overhead = sizer.asizeof(self)
546  if snapshot.asizeof_total:
547  snapshot.asizeof_total -= snapshot.overhead
548 
549  self.snapshots.append(snapshot)
550 
551  finally:
552  self.snapshot_lock.release()
