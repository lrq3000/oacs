17 def _merge_asized(base, other, level=0):
18  """
19  Merge **Asized** instances `base` and `other` into `base`.
20  """
21  ref2key = lambda ref: ref.name.split(':')[0]
22  base.size += other.size
23  base.flat += other.flat
24  if level > 0:
25  base.name = ref2key(base)
26  # Add refs from other to base. Any new refs are appended.
27  base.refs = list(base.refs) # we may need to append items
28  refs = {}
29  for ref in base.refs:
30  refs[ref2key(ref)] = ref
31  for ref in other.refs:
32  key = ref2key(ref)
33  if key in refs:
34  _merge_asized(refs[key], ref, level=level+1)
35  else:
36  # Don't modify existing Asized instances => deepcopy
37  base.refs.append(deepcopy(ref))
38  base.refs[-1].name = key

155  def sort_stats(self, *args):
156  """
157  Sort the tracked objects according to the supplied criteria. The
158  argument is a string identifying the basis of a sort (example: 'size'
159  or 'classname'). When more than one key is provided, then additional
160  keys are used as secondary criteria when there is equality in all keys
161  selected before them. For example, ``sort_stats('name', 'size')`` will
162  sort all the entries according to their class name, and resolve all
163  ties (identical class names) by sorting by size. The criteria are
164  fields in the tracked object instances. Results are stored in the
165  ``self.sorted`` list which is used by ``Stats.print_stats()`` and other
166  methods. The fields available for sorting are:
167 
168  'classname'
169  the name with which the class was registered
170  'name'
171  the classname
172  'birth'
173  creation timestamp
174  'death'
175  destruction timestamp
176  'size'
177  the maximum measured size of the object
178  'tsize'
179  the measured size during the largest snapshot
180  'repr'
181  string representation of the object
182 
183  Note that sorts on size are in descending order (placing most memory
184  consuming items first), whereas name, repr, and creation time searches
185  are in ascending order (alphabetical).
186 
187  The function returns self to allow calling functions on the result::
188 
189  stats.sort_stats('size').reverse_order().print_stats()
190  """
191 
192  criteria = ('classname', 'tsize', 'birth', 'death',
193  'name', 'repr', 'size')
194 
195  if not set(criteria).issuperset(set(args)):
196  raise ValueError("Invalid sort criteria")
197 
198  if not args:
199  args = criteria
200 
201  def args_to_tuple(obj):
202  keys = []
203  for attr in args:
204  attribute = getattr(obj, attr)
205  if attr in ('tsize', 'size'):
206  attribute = -attribute
207  keys.append(attribute)
208  return tuple(keys)
209 
210  self._init_sort()
211  self.sorted.sort(key=args_to_tuple)
212 
213  return self
