69 def get_ref(obj):
70  """
71  Get string reference to object. Stores a weak reference in a dictionary
72  using the object's id as the key. If the object cannot be weakly
73  referenced (e.g. dictionaries, frame objects), store a strong references
74  in a classic dictionary.
75 
76  Returns the object's id as a string.
77  """
78  oid = id(obj)
79  try:
80  server.id2ref[oid] = obj
81  except TypeError:
82  server.id2obj[oid] = obj
83  return str(oid)

171 def get_traceback(threadid):
172  threadid = int(threadid)
173  frames = sys._current_frames()
174  if threadid in frames:

277 def start_profiler(host='localhost', port=8090, tracker=None, stats=None,
278  debug=False, **kwargs):
279  """
280  Start the web server to show profiling data. The function suspends the
281  Python application (the current thread) until the web server is stopped.
282 
283  The only way to stop the server is to signal the running thread, e.g. press
284  Ctrl+C in the console. If this isn't feasible for your application use
285  `start_in_background` instead.
286 
287  During the execution of the web server, profiling data is (lazily) cached
288  to improve performance. For example, garbage graphs are rendered when the
289  garbage profiling data is requested and are simply retransmitted upon later
290  requests.
291 
292  :param host: the host where the server shall run, default is localhost
293  :param port: server listens on the specified port, default is 8090 to allow
294  coexistance with common web applications
295  :param tracker: `ClassTracker` instance, browse profiling data (on-line
296  analysis)
297  :param stats: `Stats` instance, analyze `ClassTracker` profiling dumps
298  (useful for off-line analysis)
299  """
300  if tracker and not stats:
301  server.stats = tracker.stats
302  else:
303  server.stats = stats
304  try:
305  server.tmpdir = mkdtemp(prefix='pympler')
306  server.server = PymplerServer(host=host, port=port, **kwargs)
307  bottle.debug(debug)
308  bottle.run(server=server.server)
309  finally:
310  rmtree(server.tmpdir)
