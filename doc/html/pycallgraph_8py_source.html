126 class GlobbingFilter(object):

133  def __init__(self, include=None, exclude=None, max_depth=None,
134  min_depth=None):
135  if include is None and exclude is None:
136  include = ['*']
137  exclude = []
138  elif include is None:
139  include = ['*']
140  elif exclude is None:
141  exclude = []

143  self.exclude = exclude
144  if max_depth is None:
145  self.max_depth = max_depth or 9999
146  else:
147  self.max_depth = max_depth

151  self.min_depth = min_depth or 0

153  def __call__(self, stack, module_name=None, class_name=None,
154  func_name=None, full_name=None):
155  from fnmatch import fnmatch
156  if len(stack) > self.max_depth:
157  return False
158  if len(stack) < self.min_depth:
159  return False
160  for pattern in self.exclude:
161  if fnmatch(full_name, pattern):
162  return False
163  for pattern in self.include:
164  if fnmatch(full_name, pattern):
165  return True
166  return False

211 def tracer(frame, event, arg):
212  """This is an internal function that is called every time a call is made
213  during a trace. It keeps track of relationships between calls.
214  """
215  global func_count_max
216  global func_count
217  global trace_filter
218  global time_filter
219  global call_stack
220  global func_time
221  global func_time_max
222 
223  if event == 'call':
224  keep = True
225  code = frame.f_code
226 
227  # Stores all the parts of a human readable name of the current call.
228  full_name_list = []
229 
230  # Work out the module name
231  module = inspect.getmodule(code)
232  if module:
233  module_name = module.__name__
234  module_path = module.__file__
235  if not settings['include_stdlib'] \
236  and is_module_stdlib(module_path):
237  keep = False
238  if module_name == '__main__':
239  module_name = ''
240  else:
241  module_name = ''
242  if module_name:
243  full_name_list.append(module_name)
244 
245  # Work out the class name.
246  try:
247  class_name = frame.f_locals['self'].__class__.__name__
248  full_name_list.append(class_name)
249  except (KeyError, AttributeError):
250  class_name = ''
251 
252  # Work out the current function or method
253  func_name = code.co_name
254  if func_name == '?':
255  func_name = '__main__'
256  full_name_list.append(func_name)
257 
258  # Create a readable representation of the current call
259  full_name = '.'.join(full_name_list)
260 
261  # Load the trace filter, if any. 'keep' determines if we should ignore
262  # this call
263  if keep and trace_filter:
264  keep = trace_filter(call_stack, module_name, class_name,
265  func_name, full_name)
266 
267  # Store the call information
268  if keep:
269 
270  if call_stack:
271  fr = call_stack[-1]
272  else:
273  fr = None
274  if fr not in call_dict:
275  call_dict[fr] = {}
276  if full_name not in call_dict[fr]:
277  call_dict[fr][full_name] = 0
278  call_dict[fr][full_name] += 1
279 
280  if full_name not in func_count:
281  func_count[full_name] = 0
282  func_count[full_name] += 1
283  if func_count[full_name] > func_count_max:
284  func_count_max = func_count[full_name]
285 
286  call_stack.append(full_name)
287  call_stack_timer.append(time.time())
288 
289  else:
290  call_stack.append('')
291  call_stack_timer.append(None)
292 
293  if event == 'return':
294  if call_stack:
295  full_name = call_stack.pop(-1)
296  if call_stack_timer:
297  t = call_stack_timer.pop(-1)
298  else:
299  t = None
300  if t and time_filter(stack=call_stack, full_name=full_name):
301  if full_name not in func_time:
302  func_time[full_name] = 0
303  call_time = (time.time() - t)
304  func_time[full_name] += call_time
305  if func_time[full_name] > func_time_max:
306  func_time_max = func_time[full_name]
307 
308  return tracer

470 def simple_memoize(callable_object):
471  """Simple memoization for functions without keyword arguments.
472 
473  This is useful for mapping code objects to module in this context.
474  inspect.getmodule() requires a number of system calls, which may slow down
475  the tracing considerably. Caching the mapping from code objects (there is
476  *one* code object for each function, regardless of how many simultaneous
477  activations records there are).
478 
479  In this context we can ignore keyword arguments, but a generic memoizer
480  ought to take care of that as well.
481  """
482 
483  cache = dict()
484 
485  def wrapper(*rest):
486  if rest not in cache:
487  cache[rest] = callable_object(*rest)
488  return cache[rest]
489 
490  return wrapper
