1 """A collection of functions to summarize object information.
2 
3 This module provides several function which will help you to analyze object
4 information which was gathered. Often it is sufficient to work with aggregated
5 data instead of handling the entire set of existing objects. For example can a
6 memory leak identified simple based on the number and size of existing objects.
7 
8 A summary contains information about objects in a table-like manner. Technically,
9 it is a list of lists. Each of these lists represents a row, whereas the first
10 column reflects the object type, the second column the number of objects, and
11 the third column the size of all these objects. This allows a simple table-like
12 output like the following:
13 
14 ============= ============ =============
15  types # objects total size
16 ============= ============ =============
17 <type 'dict'> 2 560
18  <type 'str'> 3 126
19  <type 'int'> 4 96
20 <type 'long'> 2 66
21 <type 'list'> 1 40
22 ============= ============ =============
23 
24 Another advantage of summaries is that they influence the system you analyze
25 only to a minimum. Working with references to existing objects will keep these
26 objects alive. Most of the times this is no desired behavior (as it will have
27 an impact on the observations). Using summaries reduces this effect greatly.
28 
29 output representation
30 ---------------------
31 
32 The output representation of types is defined in summary.representations.
33 Every type defined in this dictionary will be represented as specified. Each
34 definition has a list of different representations. The later a representation
35 appears in this list, the higher its verbosity level. From types which are not
36 defined in summary.representations the default str() representation will be
37 used.
38 
39 Per default, summaries will use the verbosity level 1 for any encountered type.
40 The reason is that several computations are done with summaries and rows have to
41 remain comparable. Therefore information which reflect an objects state,
42 e.g. the current line number of a frame, should not be included. You may add
43 more detailed information at higher verbosity levels than 1.
44 """

112 def summarize(objects):
113  """Summarize an objects list.
114 
115  Return a list of lists, whereas each row consists of::
116  [str(type), number of objects of this type, total size of these objects].
117 
118  No guarantee regarding the order is given.
119 
120  """
121  count = {}
122  total_size = {}
123  for o in objects:
124  otype = _repr(o)
125  if otype in count:
126  count[otype] += 1
127  total_size[otype] += _getsizeof(o)
128  else:
129  count[otype] = 1
130  total_size[otype] = _getsizeof(o)
131  rows = []
132  for otype in count:
133  rows.append([otype, count[otype], total_size[otype]])
134  return rows

136 def get_diff(left, right):
137  """Get the difference of two summaries.
138 
139  Subtracts the values of the right summary from the values of the left
140  summary.
141  If similar rows appear on both sides, the are included in the summary with
142  0 for number of elements and total size.
143  If the number of elements of a row of the diff is 0, but the total size is
144  not, it means that objects likely have changed, but not there number, thus
145  resulting in a changed size.
146 
147  """
148  res = []
149  for row_r in right:
150  found = False
151  for row_l in left:
152  if row_r[0] == row_l[0]:
153  res.append([row_r[0], row_r[1] - row_l[1], row_r[2] - row_l[2]])
154  found = True
155  if not found:
156  res.append(row_r)
157 
158  for row_l in left:
159  found = False
160  for row_r in right:
161  if row_l[0] == row_r[0]:
162  found = True
163  if not found:
164  res.append([row_l[0], -row_l[1], -row_l[2]])
165  return res

204 def _print_table(rows, header=True):
205  """Print a list of lists as a pretty table.
206 
207  Keyword arguments:
208  header -- if True the first row is treated as a table header
209 
210  inspired by http://aspn.activestate.com/ASPN/Cookbook/Python/Recipe/267662
211  """
212  border = "="
213  # vertical delimiter
214  vdelim = " | "
215  # padding nr. of spaces are left around the longest element in the
216  # column
217  padding = 1
218  # may be left,center,right
219  justify = 'right'
220  justify = {'left' : str.ljust,
221  'center' : str.center,
222  'right' : str.rjust}[justify.lower()]
223  # calculate column widths (longest item in each col
224  # plus "padding" nr of spaces on both sides)
225  cols = zip(*rows)
226  colWidths = [max([len(str(item))+2*padding for item in col]) for col in cols]
227  borderline = vdelim.join([w*border for w in colWidths])
228  for row in rows:

240 def _repr(o, verbosity=1):
241  """Get meaning object representation.
242 
243  This function should be used when the simple str(o) output would result in
244  too general data. E.g. "<type 'instance'" is less meaningful than
245  "instance: Foo".
246 
247  Keyword arguments:
248  verbosity -- if True the first row is treated as a table header
249 
250  """
251  res = ""
252 
253  t = type(o)
254  if (verbosity == 0) or (t not in representations):
255  res = str(t)
256  else:
257  verbosity -= 1
258  if len(representations[t]) < verbosity:
259  verbosity = len(representations[t]) - 1
260  res = representations[t][verbosity](o)
261 
262  res = address.sub('', res)
263  res = type_prefix.sub('', res)
264  res = type_suffix.sub('', res)
265
