91  def _get_tree(self, root, maxdepth):
92  """Workhorse of the get_tree implementation.
93 
94  This is an recursive method which is why we have a wrapper method.
95  root is the current root object of the tree which should be returned.
96  Note that root is not of the type _Node.
97  maxdepth defines how much further down the from the root the tree
98  should be build.
99 
100  """
101  self.ignore.append(inspect.currentframe())
102  res = _Node(root, self.str_func) #PYCHOK use root parameter
103  self.already_included.add(id(root)) #PYCHOK use root parameter
104  if maxdepth == 0:
105  return res
106  objects = gc.get_referrers(root) #PYCHOK use root parameter
107  self.ignore.append(objects)
108  for o in objects:
109  # XXX: find a better way to ignore dict of _Node objects
110  if isinstance(o, dict):
111  sampleNode = _Node(1)
112  if list(sampleNode.__dict__.keys()) == list(o.keys()):
113  continue
114  _id = id(o)
115  if not self.repeat and (_id in self.already_included):
116  s = self.str_func(o)
117  res.children.append("%s (already included, id %s)" %\
118  (s, _id))
119  continue
120  if (not isinstance(o, _Node)) and (o not in self.ignore):
121  res.children.append(self._get_tree(o, maxdepth-1))
122  return res

151  def _print(self, tree, prefix, carryon):
152  """Compute and print a new line of the tree.
153 
154  This is a recursive function.
155 
156  arguments
157  tree -- tree to print
158  prefix -- prefix to the current line to print
159  carryon -- prefix which is used to carry on the vertical lines
160 
161  """
162  level = prefix.count(self.cross) + prefix.count(self.vline)
163  len_children = 0
164  if isinstance(tree , _Node):
165  len_children = len(tree.children)
166 
167  # add vertex
168  prefix += str(tree)
169  # and as many spaces as the vertex is long
170  carryon += self.space * len(str(tree))
171  if (level == self.maxdepth) or (not isinstance(tree, _Node)) or\
172  (len_children == 0):
173  self.stream.write(prefix+'\n')
174  return
175  else:
176  # add in between connections
177  prefix += self.hline
178  carryon += self.space
179  # if there is more than one branch, add a cross
180  if len(tree.children) > 1:
181  prefix += self.cross
182  carryon += self.vline
183  prefix += self.hline
184  carryon += self.space
185 
186  if len_children > 0:
187  # print the first branch (on the same line)
188  self._print(tree.children[0], prefix, carryon)
189  for b in range(1, len_children):
190  # the caryon becomes the prefix for all following children
191  prefix = carryon[:-2] + self.cross + self.hline
192  # remove the vlines for any children of last branch
193  if b == (len_children-1):
194  carryon = carryon[:-2] + 2*self.space
195  self._print(tree.children[b], prefix, carryon)
196  # leave a free line before the next branch
197  if b == (len_children-1):
198  if len(carryon.strip(' ')) == 0:
199  return
200  self.stream.write(carryon[:-2].rstrip()+'\n')
