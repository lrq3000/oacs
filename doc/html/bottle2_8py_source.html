213  match = None

215  pre, name, rex = match.groups()

450  def _cast(self, out, request, response, peek=None):
451  """ Try to convert the parameter into something WSGI compatible and set
452  correct HTTP headers when possible.
453  Support: False, str, unicode, dict, HTTPResponse, HTTPError, file-like,
454  iterable of strings and iterable of unicodes
455  """
456  # Filtered types (recursive, because they may return anything)
457  for testtype, filterfunc in self.castfilter:
458  if isinstance(out, testtype):
459  return self._cast(filterfunc(out), request, response)
460 
461  # Empty output is done here
462  if not out:
463  response.headers['Content-Length'] = 0
464  return []
465  # Join lists of byte or unicode strings. Mixed lists are NOT supported
466  if isinstance(out, list) and isinstance(out[0], (StringType, unicode)):
467  out = out[0][0:0].join(out) # b'abc'[0:0] -> b''
468  # Encode unicode strings
469  if isinstance(out, unicode):
470  out = out.encode(response.charset)
471  # Byte Strings are just returned
472  if isinstance(out, StringType):
473  response.headers['Content-Length'] = str(len(out))
474  return [out]
475  # HTTPError or HTTPException (recursive, because they may wrap anything)
476  if isinstance(out, HTTPError):
477  out.apply(response)
478  return self._cast(self.error_handler.get(out.status, repr)(out), request, response)
479  if isinstance(out, HTTPResponse):
480  out.apply(response)
481  return self._cast(out.output, request, response)

484  if hasattr(out, 'read') and 'wsgi.file_wrapper' in request.environ:

489  try:
490  out = iter(out)

493  first = out.next()

1037  of the func parameter. This may yield more than one route if the function
1038  takes optional keyword arguments. The output is best described by example:
