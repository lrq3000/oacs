1 """
2 This module queries process memory allocation metrics from the operating
3 system. It provides a platform independent layer to get the amount of virtual
4 and physical memory allocated to the Python process.
5 
6 Different mechanisms are implemented: Either the process stat file is read
7 (Linux), the `ps` command is executed (BSD/OSX/Solaris) or the resource module
8 is queried (Unix fallback). On Windows try to use the win32 module if

82 class _ProcessMemoryInfoPS(_ProcessMemoryInfo):
83  def update(self):
84  """
85  Get virtual and resident size of current process via 'ps'.
86  This should work for MacOS X, Solaris, Linux. Returns true if it was
87  successful.
88  """
89  try:
90  p = Popen(['/bin/ps', '-p%s' % self.pid, '-o', 'rss,vsz'],
91  stdout=PIPE, stderr=PIPE)
92  except OSError: # pragma: no cover
93  pass
94  else:
95  s = p.communicate()[0].split()
96  if p.returncode == 0 and len(s) >= 2: # pragma: no branch
97  self.vsz = int(s[-1]) * 1024
98  self.rss = int(s[-2]) * 1024
99  return True
100  return False # pragma: no cover

118  def update(self):
119  """
120  Get virtual size of current process by reading the process' stat file.
121  This should work for Linux.
122  """
123  try:
124  stat = open('/proc/self/stat')
125  status = open('/proc/self/status')
126  except IOError: # pragma: no cover
127  return False
128  else:
129  stats = stat.read().split()
130  self.vsz = int( stats[22] )
131  self.rss = int( stats[23] ) * self.pagesize
132  self.pagefaults = int( stats[11] )
133 
134  for entry in status.readlines():
135  key, value = entry.split(':')
136  size_in_bytes = lambda x: int(x.split()[0]) * 1024
137  if key == 'VmData':
138  self.data_segment = size_in_bytes(value)
139  elif key == 'VmExe':
140  self.code_segment = size_in_bytes(value)
141  elif key == 'VmLib':
142  self.shared_segment = size_in_bytes(value)
143  elif key == 'VmStk':
144  self.stack_segment = size_in_bytes(value)
145  key = self.key_map.get(key)
146  if key:
147  self.os_specific.append((key, value.strip()))
148 
149 
150  stat.close()
151  status.close()
152  return True

159  def update(self):
160  """
161  Get memory metrics of current process through `getrusage`. Only
162  available on Unix, on Linux most of the fields are not set,
163  and on BSD units are used that are not very helpful, see:
164 
165  http://www.perlmonks.org/?node_id=626693
166 
167  Furthermore, getrusage only provides accumulated statistics (e.g.
168  max rss vs current rss).
169  """
170  usage = getrusage(RUSAGE_SELF)
171  self.rss = usage.ru_maxrss * 1024
172  self.data_segment = usage.ru_idrss * 1024 # TODO: ticks?
173  self.shared_segment = usage.ru_ixrss * 1024 # TODO: ticks?
174  self.stack_segment = usage.ru_isrss * 1024 # TODO: ticks?
175  self.vsz = self.data_segment + self.shared_segment + \
176  self.stack_segment
177 
178  self.pagefaults = usage.ru_majflt
179  return self.rss != 0
